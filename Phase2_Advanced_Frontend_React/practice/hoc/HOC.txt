function App() {
  const WithBorderAlpha = hoc(Alpha);
  const WithBorderBeta = hoc(Beta);

  const AlphaWithCardLook = withCardLook(Alpha);
  const BetaWithCardLook = withCardLook(Beta);

       <WithBorderBeta/>

       <AlphaWithCardLook user="Samrat" />

}

export default App

import React from 'react'

  }
}
